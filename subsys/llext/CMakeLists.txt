  zephyr_library_compile_definitions(-D_POSIX_C_SOURCE=200809L)

		llext_load.c
		llext_link.c

		buf_loader.c
	)
  zephyr_library_sources_ifdef(CONFIG_LLEXT_SHELL shell.c)

  zephyr_library_sources(llext_export.c)
  zephyr_library_property(LATE_LINK TRUE)
